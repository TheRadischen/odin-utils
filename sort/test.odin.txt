test_stable :: proc() {
    for i in 0..<600 {
        for _ in 0..<10 {
            arr := rand_Data(i)
            copy := data_to_int(arr)
            defer delete(arr)
            blitsort(arr, greater_data)
            if !is_sorted_data(arr) {
                fmt.println(copy)
                panic("not sorted")
            }
        }
    }

    for s in sizes {
        for t in input_types {
            arr_data = t(s)
            defer delete(arr_data)

            sum := sum_data(arr_data)
            blitsort(arr_data, greater_data)
            assert(sum == sum_data(arr_data))
            assert(is_sorted_data(arr_data))
        }
    }
    fmt.println("sort is stable")
}

rand :: proc(start: u64) -> u64{
	start := start
	start ~= start << 13
	start ~= start >> 7
	start ~= start << 17
	return start
}

draw :: proc(){
        rl.BeginDrawing()

            rl.ClearBackground(rl.BLACK)
            n := len(arr)
            max_val := arr[len(arr)-1]
            for i in 0..<n {
                if arr[i] == diff[i] {
                    changed[i] = max(changed[i] - 5, -360)
                }  else {
                    diff[i] = arr[i]
                    changed[i] = 0
                }

                rl.DrawRectangle(cast(i32)((i * (GAME_SCREEN_WIDTH-20))/n) + 10, (GAME_SCREEN_HEIGHT-20)-cast(i32)(arr[i] * (GAME_SCREEN_HEIGHT-20) / max_val) + 10, 5,5, color)

counter := 0
greater :: proc(l, r: int) -> bool {
    counter += 1
    return l > r
}
greater_draw :: proc(l, r: int) -> bool {
    counter += 1
    if counter % 1000 == 0 do  draw()
    return l > r
}

greater_data :: proc(l, r: Data) -> bool {
    counter += 1
    return l.rand > r.rand
}

greater_data_draw :: proc(l, r: Data) -> bool {
    counter += 1
    if counter % 1000 == 0 do  draw()
    return l.rand > r.rand
}

nlogn :: proc(n, t: int) -> f64 {
    t := f64(t)
    n := f64(n)
    t /= linalg.log2(n) * n
    
    return t
}
